#define BOOST_TEST_MODULE MyTest

#include <boost/test/unit_test.hpp>
#include <Board.h>

int add(int i, int j) { return i + j; }

BOOST_AUTO_TEST_CASE(my_test) {
    Board board;
    // seven ways to detect and report the same error:
    int value = add(2, 2);
    BOOST_CHECK(value == 4);        // #1 continues on error

    BOOST_REQUIRE(value == 4);      // #2 throws on error

    if (value != 4)
        BOOST_ERROR("Ouch...");            // #3 continues on error

    if (value != 4)
        BOOST_FAIL("Ouch...");             // #4 throws on error

    if (value != 4) throw "Ouch..."; // #5 throws on error

    BOOST_CHECK_MESSAGE(value == 4,  // #6 continues on error
                        "add(..) result: " << value);

    BOOST_CHECK_EQUAL(value, 4);      // #7 continues on error
    Board board;
    board.sayHello();

}
